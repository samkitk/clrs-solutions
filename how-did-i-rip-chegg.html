PART 1: inject Chegg

  1. Visit chegg homework help, e.g.: https://www.chegg.com/homework-help/Introduction-to-Algorithms-3rd-edition-chapter-9.3-problem-2E-solution-9780262033848

  2. Open Chrome DevTools, add a breakpoint at webpack:///./~/redux/es/createStore.js:242 Note: You will need to log into your account first. The code may change in further Chegg releases, but the key here is to patch the dispatch function of Redux:

PART 2: Run local Node.js server

You will need a local Node.js server to write data to the disk. I will provide the POC code. To run the code, you will need a relatively new Node.js, npm i the packages and md the folder structures. The code below (1) saves ripped data to disk (2) downloads images which are part of the ripped data from AWS to disk

app.post('/collect', function (req, res, next) {
  let data = req.body;
  let chapterPath = './data/' + data.chapterName;
  if (!fs.existsSync(chapterPath)) {
    !fs.mkdirSync(chapterPath);
  }
  let problemPath = chapterPath + "/" + data.problemName;
  if (!fs.existsSync(problemPath)) {
    !fs.mkdirSync(problemPath);
  }
  let rawDataPath = chapterPath + "/" + data.problemName + "/rawdata.json";
  fs.writeFileSync(rawDataPath, JSON.stringify(data, null, 4), { encoding: "utf8" });
  let stepsPromises = [];
  data.solution.steps.forEach((step, index) => {
    stepsPromises.push(new Promise(resolveStep => {
      let stepPath = chapterPath + "/" + data.problemName + "/" + step.id + ".html";
      let myregexp = /src=["'](http.+?)['"]/img;
      let match = myregexp.exec(step.html);
      let promises = [];
      while (match != null) {
        let imageUrl = match[1];
        let tmpPath = (+new Date()) + '.' + Math.random();
        promises.push(new Promise(resolve => {
          dlqueue.add(() => new Promise(queueResolve => {
            request({
              url: imageUrl,
              maxAttempts: 5,
              retryDelay: 3000,
              encoding: null,
            }).then(function (response) {
              fs.writeFileSync(tmpPath, response.body);
              checksum.file(tmpPath, (err, sum) => {
                fs.renameSync(tmpPath, chapterPath + "/" + data.problemName + "/" + sum);
                step.html = step.html.split(imageUrl).join(sum);
                resolve();
                queueResolve();
              });
            });
          }));
        }));
        match = myregexp.exec(step.html);
      }
      Promise.all(promises).then(_ => {
        if (step.html) {
          fs.writeFileSync(stepPath, step.html, { encoding: "utf8" });
          resolveStep(step.html);

PART3: Rip Chegg

Now, using the same browser window as in PART1, run 

 
xxxxxxxxxx
ripBook();

in DevTools. It should start the automatical process of ripping.

It will stop for 5-15 seconds for each solution opened, to (1) simulate real-human browsing (2) give the Node.js server enough time to download everything.

There are also rip() and ripChapter() available in Chrome DevTools, you may try them first to make sure everything is working.

PART4: Repackage it

Now you have ripped data in ./data, but it's not very user friendly. You may use the following code to turn it into a website as I have.
